        extractParam
        		
        							
        Public Sub extractParam(source As JsonObject, idex As Integer, currParam As ConverterParam, converters List As AbstractJsonConverter)
        			        				
        Extracts a parameter to a specific index of SamsaraParamVals, using a ConverterParam and list of converters.
        	
        	idx	Argument	Data Type	Attributes	Comments
        1sourceJsonObjectJsonObject from which to extract parameters for a function 
        2idexIntegerindex of parameters into which to extract the value 
        3currParamConverterParamConverterParam object containing default value 
        4convertersAbstractJsonConverterListList of AbstractSamsaraConverters to use to convert the value 
        	
         
        Code Sample
        %REM
        The following code is from JsonSetterConverter.loadParamValuesForJsonObject.
        Loops through Me.params and constructs a corresponding Variant array of values. If the currParam's label exists in the parent JSON object (source), it is extracted. If a converter exists for that label, it will be used from the converters array. If the currParam's label does not exist in the parent JSON object, the currParam's default will be used instead.
        %END REM
        Dim currParam as CoverterParam
        Dim i as Integer
        For i = 0 to UBound(Me.params)
            Set currParam = Me.params(i)
            Select Case currParam.paramType
            Case "Parent":
                Call extractParam(source, i, currParam, converters)
            End Select
        Next
